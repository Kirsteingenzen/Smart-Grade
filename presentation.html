Implementation Phase

Core Functionality Pseudo-code

// User Authentication
FUNCTION authenticateUser(email, password):
    IF email does not end with "@carsu.edu.ph":
        RETURN error "Invalid university email"
    
    users = getStoredUsers()
    user = findUser(users, email)
    
    IF user exists AND password matches:
        setCurrentUser(user)
        redirectTo("dashboard.html")
    ELSE:
        RETURN error "Invalid credentials"

// Task Management
FUNCTION addTask(title, priority, dueDate, category):
    task = createTask(title, priority, dueDate, category)
    tasks = getCurrentUserTasks()
    tasks.push(task)
    saveTasksToStorage(tasks)
    updateTaskDisplay()

// Class Notifications
FUNCTION scheduleClassNotifications():
    classes = getCurrentUserClasses()
    FOR each class in classes:
        notificationTime = class.time - 15 minutes
        scheduleNotification(notificationTime, class.name)
